#include<iostream>
using namespace std;

const int MaxSize=20;
const int INF=0x7fffffff;
const int MIN=0xc0c0c0c0;
template<class T>
class MGraph{         
        T vertex[MaxSize];
        int visited[MaxSize];
        int  edge[MaxSize][MaxSize]={{INF}};
        int verNum,edgeNum;

    public:
        MGraph(); // 构造函数
        void Prim(int v); // Prim算法 
        int MinEdge(int *LowCost,int VexNum); // 寻找最小邻接点 
};

template<class T>
MGraph<T>::MGraph(){ //构造函数 

        cout<<"顶点的个数:";
        cin>>verNum;
//        cout<<"输入顶点的值:";
//        for(int i=0;i<verNum;i++){        
//            cin>>vertex[i];
//        }
        for(int i=0;i<verNum;i++){        
            for(int j=0;j<verNum;j++)
               edge[i][j]=INF;
        }
        cout<<"边的条数:";
        cin>>edgeNum;

        for(int i=0;i<edgeNum;i++){
            int v1,v2;
            cout<<"输入边上的编号:";
            cin>>v1>>v2 ;//边是输入了两个顶点的编号 
            cout<<"输入边上的权值:";
            cin >> edge[v1][v2];
            edge[v1][v2]=edge[v2][v1];
        }
}
template<class T>
void MGraph<T>::Prim(int v){

    int adjvex[MaxSize],lowcost[MaxSize] = {INF};
    //初始化辅助数组
    for(int i = 0; i < verNum; i++){
        lowcost[i] = edge[v][i];
        adjvex[i] = v;
    } 
    lowcost[v] = 0; //将顶点v加入集合U

    for(int i = 0; i < verNum-1; i++) { // 迭代vexNum-1次 
        int MinJ = MinEdge(lowcost,verNum); // 寻找最短边的邻接点MinJ
        cout << MinJ << " " << adjvex[MinJ] << " "  << lowcost[MinJ] << endl;
        lowcost[MinJ] = 0;  // 将MinJ加入集合U

        for(int i = 0; i < verNum; i++){ // 调整辅助数组 
            if(edge[i][MinJ] < lowcost[i]){
                lowcost[i] = edge[i][MinJ];
                adjvex[i] = MinJ; 
            }
        } 
    }  
} 
template<class T>
int MGraph<T>::MinEdge(int* LowCost, int verNum) {
    int Min = INF, Min_i = 0;
    for (int i = 0; i < verNum; i++) {
        if (LowCost[i] < Min) {
            Min = LowCost[i];
            Min_i = i;
        }
    }
    return Min_i;
}
int main(){
    MGraph<int> g;
    g.Prim(0);
}

    卸载MicrosoftSQLServer2008R2安装程序

    无法卸载，因为安装了 

    SQL Server 2008 R2 Database Engine Shared
SQL Server 2008 R2 Common Files

    但是在已安装软件中找不到这两个软件，那么解决办法是

    首先点击运行，输入cmd，进入命令行模式 然后输入WMIC

    然后执行product list 找到

    SQL Server 2008 R2 Database Engine Shared
SQL Server 2008 R2 Common Files

    二者后面分别对应：

    {36F70DEE-1EBF-4707-AFA2-E035EEAEBAA1}

    {A2122A9C-A699-4365-ADF8-68FEAC125D61}

    然后进入windos/system32目录下

    执行：

    c:\Windows\System32>msiexec.exe /x {36F70DEE-1EBF-4707-AFA2-E035EEAEBAA1}
c:\Windows\System32>msiexec.exe /x {A2122A9C-A699-4365-ADF8-68FEAC125D61}

    然后再卸载
