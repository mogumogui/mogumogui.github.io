每日一题 ： 单词搜索

题目描述：

给定一个二维网格和一个单词，找出该单词是否存在于网格中。
单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。

实例：

board =

[
[‘A’,’B’,’C’,’E’],
[‘S’,’F’,’C’,’S’],
[‘A’,’D’,’E’,’E’]
]

给定 word = “ABCCED”, 返回 true
给定 word = “SEE”, 返回 true
给定 word = “ABCB”, 返回 false

提示：

  • board 和 word 中只包含大写和小写英文字母。
  • 1 <= board.length <="200
  • 1 <= board[i].length <="200
  • 1 <= word.length <="10^3

    • 自己的源码：

    class Solution {
public:
// 设置移动坐标

    vector<pair<int, int>> directions{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    bool Find_Word(vector<vector<char>> &board, vector<vector<int>> &visited, int row, int col, string word, int n )
    {
        // 判断是否查找到最后一个字母
        if(board[row][col] != word[n])
        {
            return false;
        }
        else if(n == word.length() - 1) {
            return true;   
        }

        // 置为1 表示已经访问
        visited[row][col] = 1;

        // 设置一个变量来记录结果
        bool res = false;

        for (const auto& dir: directions)
        {
            int x = row + dir.first;
            int y = col + dir.second;

            if(x >= 0 && x < board.size() && y >= 0 && y < board[0].size() )
            {
                // 如果下一个格子没有别访问 则直接进入下一步
                if(!visited[x][y])
                {   
                    bool flag = Find_Word(board,visited,x,y,word,n+1); // 直接dfs下一个
                    if(flag)
                    {
                        res = true;
                        break;
                    }
                }
            }
        }

        visited[row][col] = 0; // 重新置为0 以便回溯

        return res;

    }

    bool exist(vector<vector<char>>& board, string word) {

        int h = board.size(), w = board[0].size();

        vector<vector<int>> visited(h, vector<int>(w));

        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                bool flag = Find_Word(board, visited, i, j, word, 0);
                if (flag) {
                    return true;
                }
            }
        }
        return false;
    }

};
