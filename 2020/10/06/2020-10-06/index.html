每日一题 ：P1657 选书

题目描述：

学校放寒假时，信息学奥赛辅导老师有1,2,3……x本书，要分给参加培训的x个人，每人只能选一本书，但是每人有两本喜欢的书。老师事先让每个人将自己喜欢的书填写在一张表上。然后根据他们填写的表来分配书本，希望设计一个程序帮助老师求出所有可能的分配方案，使每个学生都满意。

输入格式：

第一行： 一个数x
第二行： ~ 第1+x 行： 每行两个数，表示ai喜欢的书的序号

输出格式：

只有一个数：总方案数total。

自己的源码：

#include<bits/stdc++.h>
using namespace std;
int fav[10001][2],tot=0;
int n;
int flag[10001]={false};
void dfs(int num)
{
    if(num==n+1)//超过n就代表搜完了
    {
        tot++;
        return;
    }
    if(!flag[fav[num][0]])//case1
    {
        flag[fav[num][0]]=true;
        dfs(num+1);
        flag[fav[num][0]]=false;
    }
    if(!flag[fav[num][1]])//case2
    {
        flag[fav[num][1]]=true;
        dfs(num+1);
        flag[fav[num][1]]=false;
    }
    //千万别忘记回溯哦QAQ
}
int main()
{
    cin>>n;
    if(n==0)
    {
        cout<<0;
        return 0;
    }
    for(int i=1;i<=n;i++)
        cin>>fav[i][0]>>fav[i][1];
    dfs(1);
    cout<<tot;
    return 0;
}
